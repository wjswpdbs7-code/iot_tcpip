html은 주석작성시 에러가 발생할 수 있습니다.
<!--

base.html:

HTML 템플릿의 기본 구조와 공통 요소를 정의할 수 있습니다.

예를 들어, 헤더, 네비게이션 바, 푸터 등을 포함할 수 있습니다. 

부모 템플릿으로서 다른 HTML 파일에서 이 파일을 확장하여 사용할 수 있습니다.

-->
<!DOCTYPE html>                                                     <!-- HTML5 문서임을 선언합니다. -->
<html lang="ko">                                                    <!-- 언어를 한국어로 설정합니다. -->
    <head>                                                              <!-- 메타데이터(예시: 문자 인코딩, 제목, 스타일 시트 링크 등)를 포함하는 head 요소입니다. -->
        <meta charset="UTF-8">                                              <!-- 문자 인코딩을 UTF-8로 설정합니다. -->
        <title>{% block title %} Flask App {% endblock %}</title>           <!-- HTML 문서의 제목을 설정합니다. {% block title %}과 {% endblock %} 사이에 다른 템플릿에서 제목을 정의할 수 있습니다. -->
        <link rel="stylesheet" href="{{url_for('static', filename='css/style.css')}}"> <!-- CSS 스타일 시트를 링크합니다. url_for 함수를 사용하여 static 폴더 내의 css/style.css 파일의 URL을 생성합니다. -->
    </head>
    <body>                                                              <!-- 웹 페이지의 본문을 나타내는 body 요소입니다. -->
        <nav>                                                               <!-- 네비게이션 바를 나타내는 nav 요소입니다. -->
            <h2> My Flask website </h2>                                   <!-- 네비게이션 바의 제목입니다. -->      
        </nav>                                                              <!-- 네비게이션 바의 끝입니다. -->
        <main>                                                              <!-- 웹 페이지의 주요 콘텐츠를 나타내는 main 요소입니다. -->
            {% block content %} {% endblock %}                                       <!-- 다른 템플릿에서 콘텐츠를 정의할 수 있는 블록입니다. -->
        </main>                                                             <!-- 주요 콘텐츠의 끝입니다. -->
        <script src="{{ url_for('static', filename='js/script.js') }}"></script> <!-- JavaScript 파일을 링크합니다. 경로 지정 -->                                                         
    </body>                                                             <!-- 웹 페이지의 본문 끝입니다. -->                                 
</html>                                                             <!-- HTML 문서의 끝입니다. -->

<!--

index.html

index.html은 base.html을 확장하여 웹 페이지의 콘텐츠를 정의하는 템플릿입니다.

base.html에서 상속받은 구조를 유지하면서, index.html에서는 사용자 정보와 할 일 목록을 표시하는 콘텐츠를 추가합니다.

{% block %}과 {% endblock %} 사이에 콘텐츠를 작성하여 base.html에서 정의된 블록을 오버라이드할 수 있습니다.

base.html에서 한번에 하지 않는 이유는 여러 페이지에서 공통적으로 사용할 수 있는 구조와 스타일을 정의하기 위해서입니다.(상속의 개념)
-->

{% extends 'base.html' %} <!-- base.html을 확장하여 index.html이 base.html의 구조와 스타일을 상속받도록 합니다. -->

{% block title %} 홈 페이지 - {{title}} {% endblock %} <!-- base.html에서 정의된 title 블록을 오버라이드하여 페이지 제목을 "홈 페이지"로 설정합니다. -->

{% block content %}<!-- block content를 오버라이드(덮어쓰기)하여 index.html에서 콘텐츠를 정의할 수 있도록 합니다. -->
    <h1> 반갑습니다. {{ user.username }}님! </h1> <!-- 사용자 이름을 표시하는 제목입니다. {{ user.username }}는 main.py에서 전달된 user_info 딕셔너리의 username 값을 참조합니다. -->
    <p> 당신의 등급은 {{ user.level }}입니다. </p> <!-- 사용자 등급을 표시하는 단락입니다. {{ user.level }}는 main.py에서 전달된 user_info 딕셔너리의 level 값을 참조합니다. -->
    <p> 우리의 차수는 {{ chasu + 1 }}차입니다. </p> <!-- 차수를 표시하는 단락입니다. {{ chasu }}는 main.py에서 전달된 class_month 변수를 참조합니다. -->
    <h2> 오늘의 상태 </h2> 
    {% if on_class %} 
        <p> 오늘은 수업 중 </p> <!-- on_class 변수가 참일 때 표시되는 단락입니다. {{ on_class }}는 main.py에서 전달된 is_active 변수를 참조합니다. -->
    {% else %} 
        <p> 오늘은 휴강일 </p> <!-- on_class 변수가 거짓일 때 표시되는 단락입니다. {{ on_class }}는 main.py에서 전달된 is_active 변수를 참조합니다. -->
    {% endif %}

    <h2> 오늘의 할 일 </h2> <!-- 오늘의 할 일을 표시하는 제목입니다. -->
    <ul> <!-- 할 일 목록을 표시하는 순서 없는 리스트입니다. -->
        {% for item in tasks %} <!-- 반복문을 사용하여 tasks 리스트의 각 항목을 item 변수 사용하게 합니다. -->
            <li> {{ item }} </li> <!-- 각 할 일 항목을 리스트 아이템으로 표시합니다. {{ item }}는 main.py에서 전달된 items 리스트의 각 항목을 참조합니다. -->
        {% endfor %} <!-- 반복문의 끝을 나타냅니다. -->
    </ul> <!-- 할 일 목록의 끝을 나타냅니다. -->
    <button onclick="showAlert()"> 클릭 해보세요 </button> <!-- 클릭 시 showAlert() 함수를 호출하는 버튼입니다. -->
    <div class="img-box"> <!-- 이미지 박스를 나타내는 div 요소입니다. -->
        <p> 정적 이미지</p> <!-- 이미지 박스의 제목입니다. -->
        <img width="100" src="{{ url_for('static', filename='images/logo.jpg') }}"> <!-- 이미지 크기 지정, 경로 지정 -->
    </div> <!-- 이미지 박스의 끝을 나타냅니다. -->
{% endblock }
